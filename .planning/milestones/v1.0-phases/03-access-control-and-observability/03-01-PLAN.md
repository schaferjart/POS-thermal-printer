           if not api_key:

           if request.endpoint in _PUBLIC_ENDPOINTS:
               return
           provided = request.headers.get("X-Print-Key", "")
           if not hmac.compare_digest(provided, api_key):
               return error_response("Invalid or missing API key", status=401)
